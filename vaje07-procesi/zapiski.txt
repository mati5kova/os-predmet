## SISTEMSKI KLIC fork()
int pid = fork();
if(pid < 0) perror(argv[0]);
else if(pid == 0) printf("sem otrok %d s starsem %d\n", getpid(), getppid());
else printf("sem stars %d z otrokom %d", getpid(), pid);
